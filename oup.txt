OUP PROC

CMP AX, 0
JL LVL10
JMP LVL100

LVL10:
MOV BX, AX
MOV AH, 2
MOV DL, '-'
INT 21H
NEG BX
MOV AX, BX

LVL100:
MOV SZ, 0
CMP AX, 0
JE DO
LVL3:
MOV BX, 0
MOV CX, 0
MOV DX, 0
CMP AX, 0
JE LVL5
INC SZ
MOV CL, 10
DIV CL
MOV CL, AH
ADD CL, '0'
MOV AH, 0
MOV BX, AX
PUSH CX
MOV AX, BX
JMP LVL3
LVL5:
MOV BX, SZ
CMP BX, 0
JE GO
POP CX
MOV AH, 2
MOV DL, CL
INT 21H
DEC SZ
JMP LVL5
DO:
MOV AH, 2
MOV DL, '0'
INT 21H
GO:
RET

OUP ENDP
